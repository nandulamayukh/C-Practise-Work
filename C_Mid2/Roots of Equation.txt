#include<stdio.h>
#include<math.h>
main()
{
int a,b,c,d,r1,r2;
printf("Please enter values for a,b and c, such that the quadratic equation follows ax^2+bx+c=0");
getch();
printf("\na=");
scanf("%d",&a);
printf("\nb=");
scanf("%d",&b);
printf("\nc=");
scanf("%d",&c);
d=(pow(b,2)-4*a*c);
if(d>=0)
{
r1=(-b+sqrt(d))/2*a;
r2=(-b-sqrt(d))/2*a;
printf("\nRoot1=%d\nRoot2=%d",r1,r2);
}
else
{
printf("Equation does not have real roots");
}
getch();
}
